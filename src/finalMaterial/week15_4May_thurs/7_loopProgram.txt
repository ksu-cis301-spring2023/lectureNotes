def diff(big: Z, small: Z): Z = {
    l"""{
        requires big >= small
        ensures result == big-small
    }"""

    //PREMISES:



    var count: Z = 0
    var cur: Z = big

    //PREMISES:



    //SHOW:



    while (cur != small) {
        l"""{
            invariant
                count == (big-cur)
            modifies count, cur
        }"""

        //PREMISES:



        cur = cur - 1
        count = count + 1

        //SHOW:


    }

    //PREMISES:



    //SHOW:



    return count
}
